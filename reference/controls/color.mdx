## Params

<ParamField query="label" type="string" default="Text">
  Text for the color label. The label appears in the panel in the Makeswift builder.
</ParamField>

<ParamField query="labelOrientation" type="LabelOrientation" default="horizontal">
  Position for the color label within the panel. 
</ParamField>

<ParamField query="defaultValue" type="string" default="undefined">
  Defines the color shown in the Makeswift Builder when the value isn't set. Can be any CSS color string. If you don't provide a default value, the control passes an `undefined` value to your component.
</ParamField>

<ParamField query="hideAlpha" type="boolean" default="false">
  Indicates whether to hide the <Tooltip tip="UI component for selecting color transparency">alpha channel slider</Tooltip>.
</ParamField>

## Types

```javascript LabelOrientation
type LabelOrientation = 'vertical' | 'horizontal'
```

<RequestExample>

```jsx /components/Button/Button.makeswift.js
import { Color } from "@makeswift/runtime/controls"
import { runtime } from "@/lib/makeswift/runtime"

import { Button } from "./Button"

runtime.registerComponent(Button, {
  type: "button",
  label: "Button",
  props: {
    backgroundColor: Color({ label: 'Color', defaultValue: 'black' }),
    color: Color({ label: 'Text color', defaultValue: 'white' })
    }
  },
)
```
